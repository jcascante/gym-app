# TypeScript/React Frontend - Copilot Instructions

> Specialized guidelines for the Next.js frontend. Reference these when generating TypeScript/React code in `apps/frontend/`.

---

## Context

- **Location**: `apps/frontend/`
- **Framework**: Next.js 15 (App Router)
- **Language**: TypeScript 5.x (strict mode)
- **React**: React 19 (functional components + hooks)
- **Type Checking**: tsc --strict (mandatory)
- **Linting**: ESLint + Prettier
- **Testing**: vitest + React Testing Library + Playwright (E2E)
- **Minimum Coverage**: 80%

## Core Principles

### 1. Strict TypeScript Always
No `any` types. Use `unknown` and type guards if needed.

```typescript
// ✅ GOOD - Fully typed
type PlanProps = {
  plan: Plan;
  onUpdate: (id: string) => void;
};

export const PlanCard: React.FC<PlanProps> = ({ plan, onUpdate }) => {
  return <button onClick={() => onUpdate(plan.id)}>{plan.title}</button>;
};

// ❌ BAD - any types
function PlanCard(props: any) {
  return <button>{props.plan}</button>;
}
```

### 2. Test-First (TDD)
Write tests BEFORE implementing components.

```typescript
// Step 1: Write failing test
it("should render plan title", () => {
  render(<PlanCard plan={mockPlan} onUpdate={vi.fn()} />);
  expect(screen.getByText("My Plan")).toBeInTheDocument();
});

// Step 2: Implement component
export const PlanCard: React.FC<PlanProps> = ({ plan }) => (
  <div>{plan.title}</div>
);
```

### 3. Server vs. Client Components
Use server components by default. Mark client components explicitly.

```typescript
// ✅ GOOD - Server component (default)
export default async function DefinitionsPage() {
  const definitions = await fetch("http://api/definitions");
  return <DefinitionsList definitions={definitions} />;
}

// ✅ GOOD - Client component (explicit)
"use client";
import { useState } from "react";

export const PlanForm: React.FC = () => {
  const [loading, setLoading] = useState(false);
  // ... component
};
```

### 4. State Management
- **Server State**: TanStack Query (@tanstack/react-query)
- **Client State**: Zustand
- **Form State**: React Hook Form

```typescript
// ✅ Server state (API data)
const { data: plan } = useQuery({
  queryKey: ["plans", id],
  queryFn: () => apiClient.getPlan(id),
});

// ✅ Client state (UI state)
const sidebarOpen = useUIStore(state => state.sidebarOpen);

// ✅ Form state
const { register, handleSubmit } = useForm<FormData>({
  resolver: zodResolver(schema),
});
```

### 5. Form Validation with Zod + React Hook Form
All forms must be validated with Zod schemas.

```typescript
// Step 1: Define Zod schema
const PlanFormSchema = z.object({
  programId: z.string().min(1, "Required"),
  athleteLevel: z.enum(["beginner", "intermediate", "advanced"]),
  weeks: z.number().int().positive().max(52),
});

type PlanFormData = z.infer<typeof PlanFormSchema>;

// Step 2: Use in component with React Hook Form
"use client";
export const PlanForm: React.FC = () => {
  const { register, handleSubmit, formState: { errors } } = useForm<PlanFormData>({
    resolver: zodResolver(PlanFormSchema),
  });

  return (
    <form onSubmit={handleSubmit(onSubmit)}>
      <input {...register("programId")} />
      {errors.programId && <span>{errors.programId.message}</span>}
    </form>
  );
};
```

### File Naming
- **Components**: `PascalCase.tsx` (e.g., `PlanCard.tsx`)
- **Hooks**: `useFeature.ts` (always starts with `use`)
- **Stores**: `featureStore.ts`
- **Utilities**: `camelCase.ts` (e.g., `apiClient.ts`)
- **Types**: Inline in files or `types.ts`
- **Tests**: `Feature.test.tsx` (colocated) or `test_feature.ts`

### Import Path Aliases
```typescript
// Use @/ prefix for imports
import { apiClient } from "@/lib/api-client";
import { Plan } from "@/lib/types";
import { PlanCard } from "@/components/plan/PlanCard";
import { useGetPlan } from "@/hooks/useGetPlan";
```

## Patterns to Follow

### Functional Component with Props
```typescript
type PlanCardProps = {
  plan: Plan;
  readonly: boolean;
  onSelect?: (id: string) => void;
};

export const PlanCard: React.FC<PlanCardProps> = ({
  plan,
  readonly,
  onSelect,
}) => {
  return (
    <div className="p-4 border rounded">
      <h3>{plan.title}</h3>
      <p>{plan.duration} weeks</p>
      {!readonly && (
        <button onClick={() => onSelect?.(plan.id)}>Select</button>
      )}
    </div>
  );
};
```

### Custom Hook with TanStack Query
```typescript
import { useQuery } from "@tanstack/react-query";
import { apiClient } from "@/lib/api-client";
import type { Plan } from "@/lib/types";

export function useGetPlan(planId: string) {
  return useQuery<Plan, Error>({
    queryKey: ["plans", planId],
    queryFn: async () => apiClient.getPlan(planId),
    enabled: !!planId,
    staleTime: 1000 * 60 * 5, // 5 min
  });
}

// Usage
"use client";
export const PlanDetail: React.FC<{ planId: string }> = ({ planId }) => {
  const { data: plan, isLoading, error } = useGetPlan(planId);

  if (isLoading) return <LoadingSpinner />;
  if (error) return <ErrorMessage error={error} />;
  if (!plan) return <NotFound />;

  return <PlanViewer plan={plan} />;
};
```

### Zustand Store
```typescript
import { create } from "zustand";

type UIStore = {
  sidebarOpen: boolean;
  toggleSidebar: () => void;
  selectedPlanId: string | null;
  setSelectedPlanId: (id: string | null) => void;
};

export const useUIStore = create<UIStore>((set) => ({
  sidebarOpen: true,
  toggleSidebar: () => set((state) => ({ sidebarOpen: !state.sidebarOpen })),
  selectedPlanId: null,
  setSelectedPlanId: (id) => set({ selectedPlanId: id }),
}));

// Usage
"use client";
export const Sidebar: React.FC = () => {
  const { sidebarOpen, toggleSidebar } = useUIStore();

  return (
    <aside className={sidebarOpen ? "open" : "closed"}>
      <button onClick={toggleSidebar}>Toggle</button>
    </aside>
  );
};
```

### Form with Zod + React Hook Form
```typescript
"use client";
import { useForm } from "react-hook-form";
import { zodResolver } from "@hookform/resolvers/zod";
import { PlanFormSchema, type PlanFormData } from "@/lib/validation";
import { useCreatePlan } from "@/hooks/useCreatePlan";

export const PlanForm: React.FC = () => {
  const createPlan = useCreatePlan();
  const {
    register,
    handleSubmit,
    formState: { errors, isSubmitting },
  } = useForm<PlanFormData>({
    resolver: zodResolver(PlanFormSchema),
  });

  return (
    <form onSubmit={handleSubmit((data) => createPlan.mutateAsync(data))}>
      <div>
        <label htmlFor="programId">Program</label>
        <select id="programId" {...register("programId")}>
          <option value="">Select...</option>
        </select>
        {errors.programId && <span>{errors.programId.message}</span>}
      </div>

      <button type="submit" disabled={isSubmitting}>
        {isSubmitting ? "Loading..." : "Generate"}
      </button>
    </form>
  );
};
```

### Test Component
```typescript
import { render, screen, fireEvent } from "@testing-library/react";
import { describe, it, expect, vi } from "vitest";
import { PlanCard } from "./PlanCard";

describe("PlanCard", () => {
  it("should render plan title", () => {
    const mockPlan = { id: "1", title: "Strength", duration: 12 };
    render(<PlanCard plan={mockPlan} readonly={false} onSelect={vi.fn()} />);

    expect(screen.getByText("Strength")).toBeInTheDocument();
  });

  it("should call onSelect when button clicked", () => {
    const onSelect = vi.fn();
    render(<PlanCard plan={mockPlan} readonly={false} onSelect={onSelect} />);

    fireEvent.click(screen.getByRole("button", { name: /select/i }));
    expect(onSelect).toHaveBeenCalledWith("1");
  });
});
```

## Testing Requirements

### Test Organization
- **Unit Tests**: Components, hooks, utilities in `tests/components/`, `tests/lib/`
- **E2E Tests**: User workflows in `tests/e2e/` (Playwright)
- **Mocking**: API responses in `tests/__mocks__/`

### Test Coverage
- **Minimum**: 80% line coverage
- **Target**: 85-90% for components and utilities
- Run: `npm run test` or `npm run test -- --coverage`

### Test Types
| Type | Tool | Use Case |
|------|------|----------|
| Unit | vitest | Components, hooks, utilities |
| Component | vitest + RTL | React component behavior |
| Integration | vitest + RTL | Multiple components together |
| E2E | Playwright | Full user workflows |

### Always Test
- ✅ Rendering (text, elements visible)
- ✅ User interactions (clicks, form fills)
- ✅ API integration (data loading, errors)
- ✅ Loading and error states
- ✅ Conditional rendering

## Styling

### Tailwind CSS + shadcn/ui
Use Tailwind classes exclusively, no inline styles.

```typescript
// ✅ GOOD - Tailwind classes
<div className="flex items-center justify-between gap-4 p-6 bg-blue-50 rounded-lg border">
  <h2 className="text-lg font-bold text-gray-900">{title}</h2>
  <button className="px-4 py-2 bg-blue-600 text-white rounded hover:bg-blue-700">
    Action
  </button>
</div>

// ❌ BAD - Inline styles
<div style={{ display: "flex", padding: "24px" }}>
  {/* ... */}
</div>
```

### shadcn/ui Components
Use pre-built components from shadcn/ui:

```typescript
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Card } from "@/components/ui/card";

export const MyComponent = () => (
  <Card>
    <Input placeholder="Enter text..." />
    <Button>Submit</Button>
  </Card>
);
```

## Error Handling & Loading States

```typescript
// ✅ GOOD - Explicit error handling
"use client";
import { useQuery } from "@tanstack/react-query";

export const PlanDetail: React.FC<{ id: string }> = ({ id }) => {
  const { data: plan, isLoading, error } = useQuery({
    queryKey: ["plans", id],
    queryFn: () => apiClient.getPlan(id),
  });

  if (isLoading) return <LoadingSpinner />;
  if (error) return <ErrorBoundary error={error} />;
  if (!plan) return <NotFound />;

  return <PlanRenderer plan={plan} />;
};
```

Never hide errors or loading states:

```typescript
// ❌ BAD - Silent failure
{plan && <PlanRenderer plan={plan} />}

// ❌ BAD - No loading state
<PlanRenderer plan={plan ?? {}} />
```

## What NOT to Do

❌ Skip tests or write tests after code
❌ Use `any` type in TypeScript
❌ Mix server and client component concerns
❌ Hardcode API URLs or credentials
❌ Ignore error states or loading states
❌ Use `console.log()` instead of proper logging
❌ Forget to memoize expensive components
❌ Create prop drilling chains (use stores instead)
❌ Use `||` for boolean checks (use `??` for nullish checks)
